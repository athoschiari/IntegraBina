Setagens (em JavaScript):
// Caminho completo do PlugIn
PathPlugIn = "C:\\PlugInRepeater\\PlugInBraiDRepeater.exe";
 
// Seu número de registro (deixe em branco se não possuir - DEMO)
RegNum = "SEU_NUMERO_DE_REGISTRO";
 
// IP da máquina onde o PlugIn foi instalado (Nesta máquina ou na rede. Só use a opção "Ativar PlugIn" se ele
 estiver nesta máquina e se já não estiver ativo)
IP = "192.168.2.2";
 
// Setagem padrão para a Linha 1
BraiDRepeaterX1.a4Remote_Port = 3821;
BraiDRepeaterX1.a5Local_Port = 6511;
BraiDRepeaterX1.a3Remote_IP = IP;
document.FOut.TPathPlugIn.value = PathPlugIn + " " +RegNum ;
 
Ativar o PlugIn
BraiDRepeaterX1.StartPlugIn(PathPlugIn, RegNum);
